<script>
let questions = [
  {q:"‡¶™‡ßÉ‡¶•‡¶ø‡¶¨‡ßÄ‡¶™‡ßÉ‡¶∑‡ßç‡¶†‡ßá‡¶∞ ‡¶ï‡¶§ ‡¶≠‡¶æ‡¶ó ‡¶™‡¶æ‡¶®‡¶ø?", o:["60%","70%","75%","85%"], a:1},
  {q:"‡¶∂‡¶∞‡ßç‡¶ï‡¶∞‡¶æ ‡¶•‡ßá‡¶ï‡ßá ‡¶∂‡¶ï‡ßç‡¶§‡¶ø?", o:["2 kcal","4 kcal","6 kcal","9 kcal"], a:1}
];

let index=0, score=0, user="";

function startQuiz(){
  user=document.getElementById("username").value;
  if(!user) return alert("‡¶®‡¶æ‡¶Æ ‡¶¶‡¶ø‡¶®");
  loginBox.style.display="none";
  quizBox.style.display="block";
  loadQuestion();
}

function loadQuestion(){
  let q=questions[index];
  qText.innerText=(index+1)+". "+q.q;
  options.innerHTML="";
  q.o.forEach((op,i)=>{
    let div=document.createElement("div");
    div.className="option";
    div.innerText=op;
    div.onclick=()=>{ if(i===q.a) score++; nextQuestion(); };
    options.appendChild(div);
  });
}

function nextQuestion(){
  index++;
  if(index<questions.length) loadQuestion();
  else showResult();
}

function showResult(){
  quizBox.style.display="none";
  resultBox.style.display="block";
  scoreText.innerText=`${user}, ‡¶Ü‡¶™‡¶®‡¶æ‡¶∞ ‡¶∏‡ßç‡¶ï‡ßã‡¶∞: ${score}/${questions.length}`;
  saveLeaderboard();
}

function saveLeaderboard(){
  let data=JSON.parse(localStorage.getItem("board")||"[]");
  data.push({name:user,score});
  localStorage.setItem("board",JSON.stringify(data));
  renderBoard();
}

function renderBoard(){
  let data=JSON.parse(localStorage.getItem("board")||"[]");
  leaderboard.innerHTML="";
  data.sort((a,b)=>b.score-a.score).forEach(d=>{
    let li=document.createElement("li");
    li.innerText=`${d.name} ‚Äî ${d.score}`;
    leaderboard.appendChild(li);
  });
}
renderBoard();

function showCertificate(){
  certificate.style.display="block";
  certName.innerText=user;
  certScore.innerText="Score: "+score+"/"+questions.length;
}
